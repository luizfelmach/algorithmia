
\begin{align*}
    (x+y) \text{mod}\ m & = (x \text{mod}\ m + y \text{mod}\ m) \text{mod}\ m \\
    (x-y) \text{mod}\ m & = (x \text{mod}\ m - y \text{mod}\ m) \text{mod}\ m \\
    (x\cdot y) \text{mod}\ m & = (x \text{mod}\ m \cdot y \text{mod}\ m) \text{mod}\ m \\
    (x^n) \text{mod}\ m & = (x \text{mod}\ m)^n  \text{mod}\ m
\end{align*}
